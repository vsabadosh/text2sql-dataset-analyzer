HOW TO RUN
==========

1. Run the pipeline:
   text2sql run --config configs/pipeline.comprehensive.yaml

2. Check the output directory:
   comprehensive_test_<timestamp>/
   ├── metrics.duckdb           ← All metrics in DuckDB
   ├── analysis_report.md       ← Auto-generated report
   ├── annotatedOutputDataset.jsonl
   └── *_metrics.jsonl files

3. View the report:
   cat comprehensive_test_*/analysis_report.md

4. Query metrics directly:
   duckdb comprehensive_test_*/metrics.duckdb

   Example queries:
   
   -- Antipattern frequency
   SELECT 
     SUM(has_select_star) as select_star,
     SUM(has_implicit_join) as implicit_join,
     SUM(has_function_in_where) as func_in_where,
     SUM(has_leading_wildcard_like) as wildcard_like,
     SUM(has_not_in_subquery) as not_in,
     SUM(has_correlated_subquery) as corr_subq,
     SUM(has_unbounded_select) as unbounded,
     SUM(has_excessive_joins) as excessive_joins
   FROM metrics_query_antipattern_annot;

   -- Complexity distribution
   SELECT 
     difficulty_level,
     COUNT(*) as count,
     AVG(complexity_score) as avg_score
   FROM metrics_query_syntax_annot
   WHERE parseable = true
   GROUP BY difficulty_level;

   -- Execution success by complexity
   SELECT 
     s.difficulty_level,
     COUNT(*) as total,
     SUM(CASE WHEN e.success THEN 1 ELSE 0 END) as passed,
     ROUND(AVG(CASE WHEN e.success THEN 1.0 ELSE 0.0 END) * 100, 1) as success_rate
   FROM metrics_query_syntax_annot s
   JOIN metrics_query_execution_annot e ON s.item_id = e.item_id
   GROUP BY s.difficulty_level;

   -- Schema issues
   SELECT 
     db_id,
     blocking_errors_total,
     fk_invalid,
     duplicate_columns_count
   FROM metrics_schema_analysis_annot
   WHERE blocking_errors_total > 0;
